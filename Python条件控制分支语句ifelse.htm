Python条件控制分支语句if…else…

#例1：if…else…语句： 
#注意：if…else…可以嵌套
n=16
if n<13:   #要有冒号
             print n   #注意空格
else:
             print "n不小于13"
#例2：if…elif…else…语句：
         n=int(raw_input())
if n<13:
                print n
elif n==13:  #注意是elif，而不是else if，注意区别
             print "n=13"
else:  
             print "n大于13"
'''条件判断语句形式：
（1）布尔表达式   非0即真！'''
if 1:
    print "true!"
else:  
    print "false!";
'''   
（2）关系表达式   >   <  ==   !=  
（3）逻辑表达式    and    or  not       '''
         ifnot 1:
               print "true!"
         else:  
             print "false!"
 
   '''      结果：
         >>>
